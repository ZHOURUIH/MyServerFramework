SCCheckPacketVersion
{

}

{
	[Optional]int mGuildID;
	[Optional]Vector<byte> mAllMission;
	[Optional]byte mGuildTitle;
	[Optional]llong mLastLoginTime;
	[Optional]llong mCreateTime;
	[Optional]int mQuickSellCount;
	[Optional]int mQuickDecomposeCount;
	[Optional]llong mPayExpiration;
	[Optional]string mQQ;
	[Optional]int mPKArenaScore;
}

SCGetItemTip[NoLog]

	Vector<NetStructItemInfo> mIDCountList;
}
